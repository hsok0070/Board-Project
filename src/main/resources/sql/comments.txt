CREATE TABLE comments(
c_cno NUMBER(10) CONSTRAINT PK_cno PRIMARY KEY,
c_bno NUMBER(10) NOT NULL CONSTRAINT FK_bno REFERENCES mw_leafboard(b_no) ON DELETE CASCADE, 
c_name VARCHAR2(20) NOT NULL,
c_content VARCHAR2(500),
c_calendar DATE DEFAULT SYSDATE
);

CREATE SEQUENCE c_cno START WITH 1 INCREMENT BY 1;
